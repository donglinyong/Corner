-- 查看名单分配是否均匀
(
select   t.type_id
        ,t.type_name
        ,t.caller_num 
        ,t.min_distribute 
        ,t.max_distribute
        ,t.total_distribute
        ,a.cnt as yes_total_distribute
    from (
        Select type_id, 
        type_name,
        count(0) as caller_num, 
        Min(cnt) as min_distribute, 
        Max(cnt) as max_distribute,
        sum(cnt) as total_distribute      
    From (
            Select type_id, type_name, caller, Count(0) As cnt
            From dm.rpt_tm_call_list_new
            Where static_date = current_date
            Group By type_id, type_name, caller
          ) x        
    Group By type_id,type_name
) t 
left join(
    select type_id,count(0) as cnt
        from dm.rpt_tm_call_list_new 
        where static_date=date_sub(current_date,interval 1 day)
    group by type_id
) a on t.type_id=a.type_id 
order by cast(t.type_id as signed) asc;

)
